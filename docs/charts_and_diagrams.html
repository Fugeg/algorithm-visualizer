interface AlgorithmState { array: ArrayElement[]; // 当前数组状态 codeLine: number; // 当前执行的代码行号 action: 'compare' | 'swap' | 'sorted'; message: string; // 操作说明 } /** * 基于生成器函数的冒泡排序状态生成器 * 使用 yield 关键字实现执行流的挂起与恢复 */ function* bubbleSortGenerator(arr: number[]): Generator<AlgorithmState> { const n = arr.length; const array = arr.map(value => ({ value, state: 'default' })); for (let i = 0; i < n - 1; i++) { let swapped = false; for (let j = 0; j < n - i - 1; j++) { // 标记比较状态 array[j].state = 'comparing'; array[j + 1].state = 'comparing'; // 生成比较状态 yield { array: [...array], codeLine: 14, action: 'compare', message: `比较元素 ${array[j].value} 和 ${array[j+1].value}` }; if (array[j].value > array[j + 1].value) { // 标记交换状态 array[j].state = 'swapping'; array[j + 1].state = 'swapping'; yield { array: [...array], codeLine: 16, action: 'swap', message: `交换元素 ${array[j].value} 和 ${array[j+1].value}` }; // 执行交换 [array[j], array[j + 1]] = [array[j + 1], array[j]]; swapped = true; } // 重置状态 array[j].state = 'default'; array[j + 1].state = 'default'; } // 标记已排序元素 array[n - i - 1].state = 'sorted'; yield { array: [...array], codeLine: 21, action: 'sorted', message: `第 ${i + 1} 轮冒泡完成` }; if (!swapped) break; } // 标记最后一个元素 array[0].state = 'sorted'; yield { array: [...array], codeLine: 26, action: 'sorted', message: '冒泡排序完成' }; }

代码2 交互控制系统播放逻辑实现

interface AnimationControllerState { isPlaying: boolean; currentStep: number; speed: number; history: AlgorithmState[]; } class AnimationController { private state: AnimationControllerState; private generator: Generator<AlgorithmState> | null = null; private animationFrameId: number | null = null; private delay: number = 500; constructor() { this.state = { isPlaying: false, currentStep: 0, speed: 1, history: [] }; } /** * 初始化算法生成器 */ initGenerator(generator: Generator<AlgorithmState>) { this.generator = generator; this.state.history = []; this.state.currentStep = 0; // 预生成所有状态 let result = generator.next(); while (!result.done) { this.state.history.push(result.value); result = generator.next(); } } /** * 播放控制 - 启动动画 */ async play() { if (this.state.isPlaying) return; this.state.isPlaying = true; this.runAnimationLoop(); } /** * 播放控制 - 暂停动画 */ pause() { this.state.isPlaying = false; if (this.animationFrameId) { cancelAnimationFrame(this.animationFrameId); } } /** * 单步前进 */ stepForward(): AlgorithmState | null { if (this.state.currentStep < this.state.history.length - 1) { this.state.currentStep++; return this.state.history[this.state.currentStep]; } return null; } /** * 单步回退 */ stepBackward(): AlgorithmState | null { if (this.state.currentStep > 0) { this.state.currentStep--; return this.state.history[this.state.currentStep]; } return null; } /** * 动画循环 - 使用 requestAnimationFrame 实现流畅动画 */ private runAnimationLoop() { const step = () => { if (!this.state.isPlaying) return; const nextState = this.stepForward(); if (nextState) { // 渲染当前状态 this.render(nextState); // 根据速度计算延迟 const adjustedDelay = this.delay / this.state.speed; this.animationFrameId = window.setTimeout(() => { requestAnimationFrame(() => this.runAnimationLoop()); }, adjustedDelay) as unknown as number; } else { // 动画结束 this.state.isPlaying = false; } }; step(); } /** * 设置播放速度 */ setSpeed(speed: number) { this.state.speed = Math.max(0.5, Math.min(4.0, speed)); } /** * 重置动画 */ reset() { this.pause(); this.state.currentStep = 0; if (this.state.history.length > 0) { this.render(this.state.history[0]); } } private render(state: AlgorithmState) { // 触发重绘，更新Canvas和代码高亮 dispatchEvent(new CustomEvent('stateChange', { detail: state })); } }
